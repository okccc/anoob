# 查看crontab运行日志：/var/log/cron*
# 获取cmd命令：cat /var/log/cron | grep "hdfs" | grep "CMD" | awk -F '(' '{print $3}' | awk -F ')' '{print $1}' | sort -u > cmd.txt
# grep "hdfs"过滤其他用户 && grep "CMD"过滤非命令行 && sort -u去重
